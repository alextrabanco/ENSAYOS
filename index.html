function rnd(min,max,dec){return Math.round((Math.random()*(max-min)+min)*Math.pow(10,dec))/Math.pow(10,dec);}

function genBrinell(){

  const D=5,t=15;

  const Fval=[100,150,200,250,300][Math.floor(Math.random()*5)];

  const d=rnd(1.5,3.5,1);

  const HB=Math.round(2*Fval/(Math.PI*D*(D-Math.sqrt(D*D-d*d)))*10)/10;

  const w=[Math.round(HB*1.22*10)/10,Math.round(HB*0.75*10)/10,Math.round(HB*1.55*10)/10];

  const opts=shuffle([HB,...w]).map(v=>v+" HB");

  return{type:"Dureza BRINELL",badge:"🔵",

    html:`<p>Ensayo de dureza <strong>BRINELL</strong>: bola <span class="dato">D = ${D} mm</span>, carga <span class="dato">F = ${Fval} kp</span>, tiempo <span class="dato">t = ${t} s</span>, huella <span class="dato">d = ${d} mm</span>.</p><p style="margin-top:8px">📐 <strong>HB = 2F / [π·D·(D − √(D²−d²))]</strong></p><p style="margin-top:4px">¿Cuál es la dureza Brinell normalizada?</p>`,

    opts,correct:opts.findIndex(o=>o===HB+" HB"),

    fb:`HB = 2×${Fval} / [π×${D}×(${D}−√(${D*D}−${d*d}))] = <strong>${HB} HB</strong> → Designación: ${HB} HB ${D} ${Fval} ${t}`};

}

function genVickers(){

  const F=[10,20,30,40,50][Math.floor(Math.random()*5)];

  const d1=rnd(0.30,0.70,2),d2=rnd(0.30,0.70,2);

  const dm=Math.round((d1+d2)/2*100)/100;

  const HV=Math.round(1.854*F/(dm*dm));

  const w=[Math.round(HV*1.18),Math.round(HV*0.82),Math.round(HV*1.4)];

  const opts=shuffle([HV,...w]).map(v=>v+" HV");

  return{type:"Dureza VICKERS",badge:"🟡",

    html:`<p>Ensayo de dureza <strong>VICKERS</strong>: diagonales <span class="dato">d₁ = ${d1} mm</span> y <span class="dato">d₂ = ${d2} mm</span>, fuerza <span class="dato">F = ${F} Kp</span>.</p><p style="margin-top:8px">📐 <strong>d_m = (d₁+d₂)/2 &nbsp;·&nbsp; HV = 1,854 × F / d_m²</strong></p>`,

    opts,correct:opts.findIndex(o=>o===HV+" HV"),

    fb:`d_m = (${d1}+${d2})/2 = ${dm} mm → HV = 1,854×${F}/${dm}² = <strong>${HV} HV</strong> → Designación: ${HV} HV ${F}`};

}

function genRockwell(){

  const tipo=Math.random()>0.5?"C":"B";

  const h1=rnd(0.05,0.15,3);

  const h3=rnd(h1+0.02,h1+0.12,3);

  const h=Math.round((h3-h1)*1000)/1000;

  const e=Math.round(h/0.002*10)/10;

  const HR=tipo==="C"?Math.round((100-e)*10)/10:Math.round((130-e)*10)/10;

  const label=tipo==="C"?"HRC":"HRB";

  const pen=tipo==="C"?"Cono de diamante (materiales duros)":"Bola de acero (materiales blandos)";

  const form=tipo==="C"?"HRC = 100 − e":"HRB = 130 − e";

  const w=[Math.round((HR+9)*10)/10,Math.round((HR-9)*10)/10,Math.round((HR+18)*10)/10];

  const opts=shuffle([HR,...w]).map(v=>v+" "+label);

  return{type:`Dureza ROCKWELL ${label}`,badge:"🔴",

    html:`<p>Ensayo <strong>ROCKWELL ${label}</strong> (${pen}): profundidad con precarga <span class="dato">h₁ = ${h1} mm</span>, tras carga y retirada <span class="dato">h₃ = ${h3} mm</span>.</p><p style="margin-top:8px">📐 <strong>h = h₃−h₁ &nbsp;·&nbsp; e = h/0,002 &nbsp;·&nbsp; ${form}</strong></p>`,

    opts,correct:opts.findIndex(o=>o===HR+" "+label),

    fb:`h = ${h3}−${h1} = ${h} mm → e = ${h}/0,002 = ${e} → ${form} = <strong>${HR} ${label}</strong>`};

}

  const KCV=Math.round(KCVmm*100*100)/100;

  const r2=shuffle([...allR2]).slice(0,20);

  const r4=shuffle([...allR4]).slice(0,15);

  const r5=generateR5();

  state={name,round:1,qIndex:0,score:0,roundScores:[0,0,0,0,0],

    r1,r2,r4,r5,log:[],r3Evaluated:false,

    matchSelected:{left:null,right:null},r3Selections:{},r3Left:null,r3Right:null,

    totalQ:r1.length+r2.length+1+r4.length+r5.length};

  showScreen('screen-game');renderQuestion();

}

function showScreen(id){document.querySelectorAll('.screen').forEach(s=>s.classList.remove('active'));document.getElementById(id).classList.add('active');}

function updateHUD(){

  let qN,qT;

  if(state.round===1){qN=state.qIndex+1;qT=state.r1.length;}

  else if(state.round===2){qN=state.qIndex+1;qT=state.r2.length;}

  else if(state.round===3){qN=1;qT=1;}

  else if(state.round===4){qN=state.qIndex+1;qT=state.r4.length;}

  else{qN=state.qIndex+1;qT=state.r5.length;}

  document.getElementById('hud-round').textContent=state.round;

  document.getElementById('hud-q').textContent=qN;

  document.getElementById('hud-total').textContent=qT;

  document.getElementById('hud-score').textContent=state.score;

  let done=0;

  if(state.round===1)done=state.qIndex;

  else if(state.round===2)done=state.r1.length+state.qIndex;

  else if(state.round===3)done=state.r1.length+state.r2.length;

  else if(state.round===4)done=state.r1.length+state.r2.length+1+state.qIndex;

  else done=state.r1.length+state.r2.length+1+state.r4.length+state.qIndex;

  document.getElementById('prog').style.width=(done/state.totalQ*100)+'%';

}

function renderQuestion(){

  updateHUD();

  else if(state.round===3)renderR3(c);

  else if(state.round===4)renderR4(c);

  else renderR5(c);

}

  const sL=shuffle([...r3Pairs]);const sR=shuffle([...r3Pairs]);

  state.r3Left=sL;state.r3Right=sR;

  state.r3Selections={};state.matchSelected={left:null,right:null};state.r3Evaluated=false;

function selectMatch(side,idx){

  if(state.r3Evaluated)return;

  const el=document.getElementById(side==='left'?`ml-${idx}`:`mr-${idx}`);

  if(state.matchSelected[side]!==null){

    const prev=document.getElementById(state.matchSelected[side]!=null?(side==='left'?`ml-${state.matchSelected[side]}`:`mr-${state.matchSelected[side]}`):'x');

    if(prev)prev.classList.remove('selected');

  }

  state.matchSelected[side]=idx;el.classList.add('selected');

  if(state.matchSelected.left!==null&&state.matchSelected.right!==null){

    const li=state.matchSelected.left,ri=state.matchSelected.right;

    if(state.r3Selections[li]!==undefined){

      const oldMr=document.getElementById(`mr-${state.r3Selections[li]}`);

      if(oldMr)oldMr.classList.remove('selected','paired');

    }

    Object.keys(state.r3Selections).forEach(k=>{

      if(state.r3Selections[k]===ri&&parseInt(k)!==li){

        const oldMl=document.getElementById(`ml-${k}`);

        if(oldMl)oldMl.classList.remove('selected','paired');

        delete state.r3Selections[k];

      }

    });

    state.r3Selections[li]=ri;

    document.getElementById(`ml-${li}`).classList.remove('selected');

    document.getElementById(`mr-${ri}`).classList.remove('selected');

    document.getElementById(`ml-${li}`).classList.add('paired');

    document.getElementById(`mr-${ri}`).classList.add('paired');

    state.matchSelected={left:null,right:null};

    if(Object.keys(state.r3Selections).length===r3Pairs.length)document.getElementById('btn-check').style.display='inline-block';

  }

}

function checkR3(){

  state.r3Evaluated=true;

  document.getElementById('btn-check').style.display='none';

  let correct=0;

  Object.keys(state.r3Selections).forEach(li=>{

    const ri=state.r3Selections[li];

    const lItem=state.r3Left[parseInt(li)];

    const rItem=state.r3Right[ri];

    const ml=document.getElementById(`ml-${li}`);

    const mr=document.getElementById(`mr-${ri}`);

    ml.classList.remove('paired');mr.classList.remove('paired');

    if(lItem.left===rItem.left){ml.classList.add('matched');mr.classList.add('matched');correct++;}

    else{ml.classList.add('wrong-match');mr.classList.add('wrong-match');}

  });

  const pts=correct*20;

  state.score+=pts;state.roundScores[2]+=pts;

  document.getElementById('hud-score').textContent=state.score;

  const allOk=correct===r3Pairs.length;

  const fb=document.getElementById('feedback');

  fb.innerHTML=`<strong>${allOk?'🎉 ¡Todo correcto!':correct>0?`✅ ${correct} de ${r3Pairs.length} correctas`:'❌ Ninguna correcta'} · +${pts} pts</strong>Recuerda: capilaridad (PT) · campo magnético (MT) · rayos X/γ (RT) · ondas mecánicas (UT) · ojo humano (VT).`;

  fb.className='feedback '+(allOk?'ok':'fail')+' show';

  state.log.push({round:3,q:'Empareja END con principio físico',ok:allOk,userAns:`${correct}/${r3Pairs.length} correctas`,correct:'5/5 correctas'});

  document.getElementById('btn-next').classList.add('show');

}

  else if(state.round===2){state.qIndex++;if(state.qIndex>=state.r2.length){state.round=3;state.qIndex=0;}}

}
